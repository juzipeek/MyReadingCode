 public  void add(Runnable job) 
{
        synchronized(jobs) {
            jobs.add(job);
            jobs.notify();
        }
}
 public boolean queueFull() 
{
        synchronized(jobs) {
            return (jobs.size()  > 0);
        }
}
 public  void run() 
{
        while (!isStopped()) {
            Runnable job;
            synchronized(jobs) {
                while (!isStopped() && jobs.size() == 0) {
                    try {
                        jobs.wait();
                    } catch (InterruptedException ex) {
                    }
                }
                if(isStopped()) break;
                job = jobs.remove(0);
            }
            job.run();
        }
}
 public synchronized  void stopWorker() 
{
        stopped = true;
        synchronized(jobs) {
            jobs.notify();
        }
